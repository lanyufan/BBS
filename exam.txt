create database exam;	--创建库

use exam;	--进库

create table Board(		--创建Board表

	Board_ID int,		--ID
	
	Board_name char(20),	--名称
	
	Board_date datetime,	--创建日期
	
	Board_describe char(64)		--描述
	
	);
	
	insert Board values(11,'Board1',2016.12.24,'描述1');
	
	insert Board values(12,'Board2',2016.12.24,'描述2');
	
create table Forum(		--创建Forum表

	Forum_FID int,		--FID
	
	Forum_name char(20),	--名称
	
	Forum_describe char(64),	--描述
	
	Board_ID int		--Forum表所属Board表的ID
	
	);
	
	insert Board values(13,'Forum1','描述1'，11);
	
	insert Board values(14,'Forum2','描述1'，12);

	);
